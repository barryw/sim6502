    }
  }

  suite("Test 6502 processor declaration") {

    }
  }

  suite("Test 6510 processor declaration") {

    test("proc-6510", "Verify 6510 declaration parses") {

    }
  }

  suite("Test default processor (no declaration)") {

    }
  }
